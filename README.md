[How To Install]


- mkdir ~/vimrc
- cd ~/vimrc
- git clone [addy]
- ln -s ~/vimrc/.vimrc ~/.vimrc
- vim
- :PluginInstall


[Install "YouCompleteMe"]
- brew install CMake
- cd ~/.vim/bundle/YouCompleteMe
- ./install.py
